Ranges chart

This jQuery plugin allows the developer to represent ranges of data into a 2D graph.

<div id="range-chart-attachment" style="width: 100%;"></div>

<script>
$(document).ready(function() {
    $('#range-chart-attachment').rangeChart({
        minValue: 60,
        maxValue: 100,
        postfix: 'Kg',
        legendDescription: 'Sneaker recommended weight',
        ranges: [
            {
                label: 'Adidas Adiboost Boston NYC 33',
                link: 'http://google.com',
                min: 70,
                max: 90
            },
            {
                label: 'Adidas Adiboost Boston NYC 33',
                link: 'http://google.com',
                min: 75,
                max: 90
            },
            {
                label: 'Adidas Adiboost Boston NYC 33',
                link: 'http://google.com',
                min: 70,
                max: 100
            },
            {
                label: 'Adidas Adiboost Boston NYC 33',
                link: 'http://google.com',
                min: 80,
                max: 95
            },
            {
                label: 'Adidas Adiboost Boston NYC 33',
                link: 'http://google.com',
                min: 65,
                max: 90
            }
        ],
        userValue: {
            value: 85,
            label: 'Your weight (85Kg)'
        }
    });
});
</script>
